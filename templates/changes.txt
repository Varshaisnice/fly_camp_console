function showGameRules(card) {
  const container = document.getElementById("rules-list-dynamic");
  container.innerHTML = ""; // clear old rules

  const rules = card.dataset.desc.split(";").map(r => r.trim()).filter(Boolean);

  rules.forEach(rule => {
    const ruleCard = document.createElement("div");
    ruleCard.className = "rule-card";
    ruleCard.textContent = rule;
    container.appendChild(ruleCard);
  });

  // show section
  document.getElementById("rules-dynamic").classList.remove("hidden");
  document.getElementById("rules-title-dynamic").textContent = `üïπÔ∏è Rules ‚Äì ${card.dataset.title}`;
}
