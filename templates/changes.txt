Below is a minimal, drop‑in patch for console.js that:

    Keeps every line you already have (no breaking changes).
    Adds a black‑screen overlay that appears only when any check fails.
    After 1 s the overlay fades out and the new page_init_status page is shown – the same list you already see on the init page, but now with the exact same OK / Failed icons.
    Retry & Back buttons on the status page (already in the HTML you added).

1. Add the overlay element once (anywhere in the <body> – after the first <div class="screen"> is fine)
html

<!-- BLACK SCREEN OVERLAY – always in the DOM -->
<div id="black-overlay" class="black-overlay"></div>

css

3. Patch console.js – only the parts that changed
js

/* ------------------------------------------------------------------ */
/* Initialisation sequence – NEW FAILURE FLOW                        */
/* ------------------------------------------------------------------ */

/* Keep the original runConnectionCheckAndStart() – we only add a tiny hook */
async function runConnectionCheckAndStart(delayStartMs = 0, skipStart = false){
  /* … existing code unchanged … */

  // ----- NEW: store results for the status page -----------------
  window.__initStoredResults = { results: results, success: !!data.success };
  window.__initReady = true;
  window.__initSuccess = !!data.success;

  if (skipStart) return;               // preview‑mode → just store
  /* … rest of the original function unchanged … */
}

/* ------------------------------------------------------------------ */
/* Show stored results on the init page (unchanged)                 */
/* ------------------------------------------------------------------ */
function showStoredInitResultsSequentially(){
  /* … unchanged … */
}

/* ------------------------------------------------------------------ */
/* NEW: FAILURE → BLACK SCREEN → STATUS PAGE                        */
/* ------------------------------------------------------------------ */
function showFailureStatusPage(){
  // 1. show black overlay
  const overlay = qs('#black-overlay');
  overlay.classList.add('active');

  // 2. after 1 s go to status page and render the same results
  setTimeout(() => {
    overlay.classList.remove('active');
    goToPage('page_init_status');
    renderStatusPage();               // ← new helper (see below)
  }, 1000);
}

/* ------------------------------------------------------------------ */
/* Render the status page – identical to the init page               */
/* ------------------------------------------------------------------ */
function renderStatusPage(){
  const stored = window.__initStoredResults;
  if (!stored) return;

  const readyEl = qs('#status-ready');
  const totalEl = qs('#status-total');
  const container = qs('#status-steps');

  totalEl.textContent = stored.results.length;
  readyEl.textContent = stored.results.filter(r => r.ok).length;
  container.innerHTML = '';

  stored.results.forEach((step, i) => {
    const row = document.createElement('div');
    row.className = `step ${step.ok ? 'ok' : 'fail'}`;
    row.innerHTML = `
      <div class="checkbox"></div>
      <span>${step.displayName} — ${step.ok ? 'Ready' : 'Failed'}</span>
    `;
    row.style.animationDelay = `${0.1 + i*0.1}s`;
    container.appendChild(row);
  });
}

/* ------------------------------------------------------------------ */
/* Hook the failure case – replace the old error‑banner logic       */
/* ------------------------------------------------------------------ */
function showStoredInitResultsSequentially(){
  const stored = window.__initStoredResults;
  if (!stored || !stored.results) return;

  clearSteps();
  const errBox = qs('#init-error');
  if (errBox) errBox.classList.add('hidden');

  qs('#init-status').textContent = 'Running connection checks...';

  let idx = 0;
  function displayNext(){
    if (idx < stored.results.length){
      const step = stored.results[idx++];
      const row = addStepRow(step.displayName);
      setTimeout(() => {
        markRow(row, step.ok, step.message);
        setTimeout(displayNext, 200);
      }, 200);
    } else {
      // ----- ALL STEPS SHOWN -----
      if (!stored.success){
        // **FAILURE** → black screen → status page
        showFailureStatusPage();
      } else {
        qs('#init-status').textContent = 'Connection OK. Preparing game...';
      }
    }
  }
  displayNext();
}

/* ------------------------------------------------------------------ */
/* Retry from the status page (already in your HTML)                */
/* ------------------------------------------------------------------ */
function retryFromStatus(){
  goToPage('page_initializing');
  setTimeout(() => {
    window.__initStarted = false;
    window.__initReady = false;
    window.__initSuccess = false;
    runConnectionCheckAndStart(3000, false);
  }, 600);
}

/* ------------------------------------------------------------------ */
/* Expose the new functions for the inline buttons                  */
/* ------------------------------------------------------------------ */
window.retryFromStatus = retryFromStatus;
