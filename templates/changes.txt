// When a game card is clicked
document.querySelectorAll('.card').forEach(card => {
  card.addEventListener('click', () => {
    const rulesSelector = card.getAttribute('data-desc');
    const rulesDiv = document.querySelector(rulesSelector);

    if (rulesDiv) {
      // Hide all rule sets
      document.querySelectorAll('.rule-set').forEach(r => r.classList.add('hidden'));
      // Show selected one
      rulesDiv.classList.remove('hidden');

      // Optional: Update title/desc
      document.getElementById('chosen-game-title').textContent = `You chose: ${card.getAttribute('data-title')}`;
      document.getElementById('chosen-game-desc').textContent = ''; // or keep empty
    }

    goToPage('page_confirm');
  });
});
